123 void update(void) {
124 currentTime = millis();
125
126 // Handle timer register rollover.
127 // The problem occurs when the current time rolls over with the target
128 // time near rollover; this results in the task stalling until the current
129 // time catches up with the target time (rare, but possible). Since
130 // the greatest difference between the targetTime and currentTime should be
131 // on the order of the longest buzzer time, if it is much greater (e.g. 100X) then
132 // that indicates a rollover has occurred.
133 if ((currentTime >= targetTime) || (targetTime - currentTime) > rolloverThreshold) {
134 if (beepData.count == 0) {
135 digitalWrite(buzzerPin, buzzerOff);
136 return;
137 }
138 // Note: to get here beepData.count must be > 0.
139 if (!beepData.doingBeepSpacing) {
140 digitalWrite(buzzerPin, buzzerOff);
141 beepData.doingBeepSpacing = true;
142 targetTime += BEEP_SPACING;
143 }
144 else {
145 digitalWrite(buzzerPin, buzzerOn);
146 beepData.doingBeepSpacing = false;
147 targetTime += beepData.duration;
148 beepData.count--;
149 }
150 }
