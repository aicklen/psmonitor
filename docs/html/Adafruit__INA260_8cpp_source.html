54 bool Adafruit_INA260::begin(uint8_t i2c_address, TwoWire *theWire) {
55  i2c_dev = new Adafruit_I2CDevice(i2c_address, theWire);
56 
57  if (!i2c_dev->begin()) {
58  return false;
59  }
60 
61  Adafruit_I2CRegister *die_register =
62  new Adafruit_I2CRegister(i2c_dev, INA260_REG_DIE_UID, 2, MSBFIRST);
63  Adafruit_I2CRegister *mfg_register =
64  new Adafruit_I2CRegister(i2c_dev, INA260_REG_MFG_UID, 2, MSBFIRST);
65  Adafruit_I2CRegisterBits *device_id =
66  new Adafruit_I2CRegisterBits(die_register, 12, 4);
67 
68  // make sure we're talking to the right chip
69  if ((mfg_register->read() != 0x5449) || (device_id->read() != 0x227)) {
70  return false;
71  }
72 
73  Config = new Adafruit_I2CRegister(i2c_dev, INA260_REG_CONFIG, 2, MSBFIRST);
74  MaskEnable =
75  new Adafruit_I2CRegister(i2c_dev, INA260_REG_MASK_ENABLE, 2, MSBFIRST);
76  AlertLimit =
77  new Adafruit_I2CRegister(i2c_dev, INA260_REG_ALERT_LIMIT, 2, MSBFIRST);
78 
79  reset();
80  delay(2); // delay 2ms to give time for first measurement to finish
81  return true;
82 }
