137 void update(void) {
138 if (!finishedFlag && executeStep) {
139 executeStep = false; // Wait for a button push to run next step
140 lcd->clear();
141 lcd->setCursor(0, 0);
142
143 // Perform calibration routine state machine
144 switch (currentState) {
145 case CALIBRATE_INITIALIZE:
146 lcd->print((const __FlashStringHelper *)promptIntro);
147 lcd->setCursor(0, 1);
148 lcd->print((const __FlashStringHelper *)promptPushButton);
149 currentState = CALIBRATE_PROMPT_V1;
150 break;
151
152 case CALIBRATE_PROMPT_V1:
153 lcd->print((const __FlashStringHelper *)promptVolts);
154 lcd->write(1);
155 lcd->print(generateVoltageString(CALIBRATE_V1));
156 lcd->setCursor(0, 1);
157 lcd->print((const __FlashStringHelper *)promptPushButton);
158 currentState = CALIBRATE_READ_V1_PROMPT_V2;
159 break;
160
161 case CALIBRATE_READ_V1_PROMPT_V2:
162 MonitorTask::getRawValues();
163 lcd->print((const __FlashStringHelper *)promptVolts);
164 lcd->write(1);
165 lcd->print(generateVoltageString(CALIBRATE_V2));
166 lcd->setCursor(0, 1);
167 lcd->print((const __FlashStringHelper *)promptPushButton);
168 currentState = CALIBRATE_READ_V2_PROMPT_I1;
169 break;
170
171 case CALIBRATE_READ_V2_PROMPT_I1:
172 MonitorTask::getRawValues();
173 lcd->print((const __FlashStringHelper *)promptCurrent);
174 (void) sprintf(string_buf, "% 5d", CALIBRATE_I1);
175 lcd->print(string_buf);
176 lcd->print(".0");
177 lcd->setCursor(0, 1);
178 lcd->print((const __FlashStringHelper *)promptPushButton);
179 currentState = CALIBRATE_READ_I1_PROMPT_I2;
180 break;
181
182 case CALIBRATE_READ_I1_PROMPT_I2:
183 MonitorTask::getRawValues();
184 lcd->print((const __FlashStringHelper *)promptCurrent);
185 (void) sprintf(string_buf, "% 5d", CALIBRATE_I2);
186 lcd->print(string_buf);
187 lcd->print(".0");
188 lcd->setCursor(0, 1);
189 lcd->print((const __FlashStringHelper *)promptPushButton);
190 currentState = CALIBRATE_READ_I2_FINISH;
191 break;
192
193 case CALIBRATE_READ_I2_FINISH:
194 default:
195 MonitorTask::getRawValues();
196 finishedFlag = true;
197 }
198 }
