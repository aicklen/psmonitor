32bool Adafruit_INA260::begin(uint8_t i2c_address, TwoWire *theWire) {
33 i2c_dev = new Adafruit_I2CDevice(i2c_address, theWire);
34
35 if (!i2c_dev->begin()) {
36 return false;
37 }
38
39 Adafruit_I2CRegister *die_register =
40 new Adafruit_I2CRegister(i2c_dev, INA260_REG_DIE_UID, 2, MSBFIRST);
41 Adafruit_I2CRegister *mfg_register =
42 new Adafruit_I2CRegister(i2c_dev, INA260_REG_MFG_UID, 2, MSBFIRST);
43 Adafruit_I2CRegisterBits *device_id =
44 new Adafruit_I2CRegisterBits(die_register, 12, 4);
45
46 // make sure we're talking to the right chip
47 if ((mfg_register->read() != 0x5449) || (device_id->read() != 0x227)) {
48 return false;
49 }
50
51 Config = new Adafruit_I2CRegister(i2c_dev, INA260_REG_CONFIG, 2, MSBFIRST);
52 MaskEnable =
53 new Adafruit_I2CRegister(i2c_dev, INA260_REG_MASK_ENABLE, 2, MSBFIRST);
54 AlertLimit =
55 new Adafruit_I2CRegister(i2c_dev, INA260_REG_ALERT_LIMIT, 2, MSBFIRST);
56
57 reset();
58 delay(2); // delay 2ms to give time for first measurement to finish
59 return true;
60}
