151 void update() {
152 currentTime = millis();
153
154 // Handle timer register rollover. Restarts task if rollover detected.
155 // The problem occurs when the current time rolls over with the target
156 // time near rollover; this results in the task stalling until the current
157 // timer catches up with the target time (rare, but possible). Since
158 // the greatest difference between the targetTime and currentTime should be
159 // on the order of the taskInterval, if it is much greater (e.g. 100X) then
160 // that indicates a rollover has occurred.
161 if (currentTime < targetTime && (targetTime - currentTime) > rolloverThreshold) {
162 targetTime = currentTime + taskInterval;
163 }
164
165 // Handle task timing. Task always runs first time through.
166 if (!startedFlag) {
167 startedFlag = true;
168 targetTime = currentTime + taskInterval;
169 }
170 else {
171 if (currentTime < targetTime) {
172 return;
173 }
174 targetTime += taskInterval;
175 }
176
177 // Process voltage readings
178 readings[MONITOR_VOLTAGE_POS] = nearest10(ina260Pos.readBusVoltageInt16());
179 readings[MONITOR_VOLTAGE_NEG] = -nearest10(ina260Neg.readBusVoltageInt16());
180
181 // Process current readings (current always treated as positive)
182 readings[MONITOR_CURRENT_POS] = ina260Pos.readCurrentInt16() + 3;
183 readings[MONITOR_CURRENT_NEG] = ina260Neg.readCurrentInt16() + 2;
184
185 // Check all voltages and current against specifications (all currents are treated as positive)
186 alert[MONITOR_VOLTAGE_POS] = (readings[MONITOR_VOLTAGE_POS] > LIMIT_MAX_VOLTAGE);
187 alert[MONITOR_VOLTAGE_NEG] = (readings[MONITOR_VOLTAGE_NEG] < -LIMIT_MAX_VOLTAGE);
188 alert[MONITOR_CURRENT_POS] = (readings[MONITOR_CURRENT_POS] > LIMIT_MAX_CURRENT);
189 alert[MONITOR_CURRENT_NEG] = (readings[MONITOR_CURRENT_NEG] > LIMIT_MAX_CURRENT);
190
191 // Alert on voltage or current out of range. Beep every OVER_RANGE_BEEP_N times
192 // the Monitor task runs.
193 if (alert[MONITOR_VOLTAGE_POS] || alert[MONITOR_VOLTAGE_NEG] || alert[MONITOR_CURRENT_POS] || alert[MONITOR_CURRENT_NEG]) {
194 if (beep_count <= 0) {
195 beep_count = OVER_RANGE_BEEP_N;
196 BuzzerTask::beep(BEEP_BLIP, 1);
197 }
198 else {
199 beep_count--;
200 }
201 }
202
203 // Display voltage data
204 lcd->setCursor(0, 0);
205 lcd->print(F("V "));
206 lcd->print(generateVoltageString(readings[MONITOR_VOLTAGE_POS]));
207 lcd->print(" ");
208 lcd->print(generateVoltageString(readings[MONITOR_VOLTAGE_NEG]));
209
210 // Display current data (note: current is always considered positive to avoid
211 // cluttering the display).
212 lcd->setCursor(0, 1);
213 lcd->print(F("mA "));
214 (void) sprintf(string_buf, "% 5d ", readings[MONITOR_CURRENT_POS]);
215 lcd->print(string_buf);
216 lcd->print(" ");
217 (void) sprintf(string_buf, "% 5d", readings[MONITOR_CURRENT_NEG]);
218 lcd->print(string_buf);
219 }
