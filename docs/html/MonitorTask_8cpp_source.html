154  void update() {
155  currentTime = millis();
156 
157  // Handle timer register rollover. Restarts task if rollover detected.
158  // The problem occurs when the current time rolls over with the target
159  // time near rollover; this results in the task stalling until the current
160  // timer catches up with the target time (rare, but possible). Since
161  // the greatest difference between the targetTime and currentTime should be
162  // on the order of the taskInterval, if it is much greater (e.g. 100X) then
163  // that indicates a rollover has occurred.
164  if (currentTime < targetTime && (targetTime - currentTime) > rolloverThreshold) {
165  targetTime = currentTime + taskInterval;
166  }
167 
168  // Handle task timing. Task always runs first time through.
169  if (!startedFlag) {
170  startedFlag = true;
171  targetTime = currentTime + taskInterval;
172  }
173  else {
174  if (currentTime < targetTime) {
175  return;
176  }
177  targetTime += taskInterval;
178  }

182  readings[MONITOR_VOLTAGE_NEG] = -nearest10( Calibration::correct(DATA_VOLTAGE_NEG, ina260Neg.readBusVoltageInt16()) );
183 
184  // Process current readings (current always treated as positive)
185  readings[MONITOR_CURRENT_POS] = Calibration::correct(DATA_CURRENT_POS, ina260Pos.readCurrentInt16());
186  readings[MONITOR_CURRENT_NEG] = Calibration::correct(DATA_CURRENT_NEG, ina260Neg.readCurrentInt16());

188  // Check all voltages and current against specifications (all currents are treated as positive)
189  alert[MONITOR_VOLTAGE_POS] = (readings[MONITOR_VOLTAGE_POS] > LIMIT_MAX_VOLTAGE);
190  alert[MONITOR_VOLTAGE_NEG] = (readings[MONITOR_VOLTAGE_NEG] < -LIMIT_MAX_VOLTAGE);
191  alert[MONITOR_CURRENT_POS] = (readings[MONITOR_CURRENT_POS] > LIMIT_MAX_CURRENT);
192  alert[MONITOR_CURRENT_NEG] = (readings[MONITOR_CURRENT_NEG] > LIMIT_MAX_CURRENT);
