125void loop() {
126 // Check mute/calibrate button; toggle mute mode if button state has
127 // transitioned from from HIGH to LOW since last check (edge triggered)
128 if (digitalRead(BUTTON_PIN) == LOW) {
129 if (previous_button_state == HIGH) {
130 previous_button_state = LOW;
131 switch(currentMode) {
132 case MODE_CALIBRATE:
133 BuzzerTask::beep(BEEP_BLIP, 1);
134 CalibrateTask::buttonPress(); // Inform the calibrate task that button was pushed
135 break;
136 case MODE_NORMAL:
137 default:
138 BuzzerTask::toggleMute();
139 break;
140 }
141 }
142 }
143 else {
144 previous_button_state = HIGH;
145 }
146
147 // Process any requested buzzer soundings
148 BuzzerTask::update();
149
150 // Dispatch to monitor or calibrate task, depending on current operational mode.
151 switch(currentMode) {
152 case MODE_CALIBRATE:
153 if (CalibrateTask::finished()) {
154 currentMode = MODE_NORMAL;
155 BuzzerTask::beep(BEEP_BLIP, 3); // Let the user know calbration is done
156 break;
157 }
158 CalibrateTask::update();
159 break;
160 case MODE_NORMAL:
161 default:
162 // Read and display voltage and current readings.
163 // This takes on the order of 15ms elapsed time.
164 MonitorTask::update();
165 break;
166 }
167 }
