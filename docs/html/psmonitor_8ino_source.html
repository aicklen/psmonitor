131 void loop() {
132  // Check mute/calibrate button; toggle mute mode if button state has
133  // transitioned from from HIGH to LOW since last check ("edge" triggered)
134  if (digitalRead(BUTTON_PIN) == LOW) {
135  if (previous_button_state == HIGH) {
136  previous_button_state = LOW;
137  switch(currentMode) {
138  case MODE_CALIBRATE:
139  BuzzerTask::beep(BEEP_BLIP, 1);
140  CalibrateTask::buttonPress(); // Inform the calibrate task that button was pushed
141  break;
142  case MODE_NORMAL:
143  default:
144  BuzzerTask::toggleMute();
145  break;
146  }

148  }
149  else {

159  break;
