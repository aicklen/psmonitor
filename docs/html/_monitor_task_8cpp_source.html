147 void update() {
148 currentTime = millis();
149
150 // Handle timer register rollover. Restarts task if rollover detected.
151 // The problem occurs when the current time rolls over with the target
152 // time near rollover; this results in the task stalling until the current
153 // timer catches up with the target time (rare, but possible). Since
154 // the greatest difference between the targetTime and currentTime should be
155 // on the order of the taskInterval, if it is much greater (e.g. 100X) then
156 // that indicates a rollover has occurred.
157 if (currentTime < targetTime && (targetTime - currentTime) > rolloverThreshold) {
158 targetTime = currentTime + taskInterval;
159 }
160
161 // Handle task timing. Task always runs first time through.
162 if (!startedFlag) {
163 startedFlag = true;
164 targetTime = currentTime + taskInterval;
165 }
166 else {
167 if (currentTime < targetTime) {
168 return;
169 }
170 targetTime += taskInterval;
171 }
172
173 // Process voltage readings
174 readings[MONITOR_VOLTAGE_POS] = nearest10(ina260Pos.readBusVoltageInt16());
175 readings[MONITOR_VOLTAGE_NEG] = -nearest10(ina260Neg.readBusVoltageInt16());
176
177 // Process current readings (current always treated as positive)
178 readings[MONITOR_CURRENT_POS] = ina260Pos.readCurrentInt16() + 3;
179 readings[MONITOR_CURRENT_NEG] = ina260Neg.readCurrentInt16() + 2;
180
181 // Check all voltages and current against specifications (all currents are treated as positive)
182 alert[MONITOR_VOLTAGE_POS] = (readings[MONITOR_VOLTAGE_POS] > LIMIT_MAX_VOLTAGE);
183 alert[MONITOR_VOLTAGE_NEG] = (readings[MONITOR_VOLTAGE_NEG] < -LIMIT_MAX_VOLTAGE);
184 alert[MONITOR_CURRENT_POS] = (readings[MONITOR_CURRENT_POS] > LIMIT_MAX_CURRENT);
185 alert[MONITOR_CURRENT_NEG] = (readings[MONITOR_CURRENT_NEG] > LIMIT_MAX_CURRENT);
186
187 // Alert on voltage or current out of range. Beep every OVER_RANGE_BEEP_N times
188 // the Monitor task runs.
189 if (alert[MONITOR_VOLTAGE_POS] || alert[MONITOR_VOLTAGE_NEG] || alert[MONITOR_CURRENT_POS] || alert[MONITOR_CURRENT_NEG]) {
190 if (beep_count <= 0) {
191 beep_count = OVER_RANGE_BEEP_N;
192 BuzzerTask::beep(BEEP_BLIP, 1);
193 }
194 else {
195 beep_count--;
196 }
197 }
198
199 // Display voltage data
200 lcd->setCursor(0, 0);
201 lcd->print(F("V "));
202 lcd->print(generateVoltageString(readings[MONITOR_VOLTAGE_POS]));
203 lcd->print(" ");
204 lcd->print(generateVoltageString(readings[MONITOR_VOLTAGE_NEG]));
205
206 // Display current data (note: current is always considered positive to avoid
207 // cluttering the display).
208 lcd->setCursor(0, 1);
209 lcd->print(F("mA "));
210 (void) sprintf(string_buf, "% 5d ", readings[MONITOR_CURRENT_POS]);
211 lcd->print(string_buf);
212 lcd->print(" ");
213 (void) sprintf(string_buf, "% 5d", readings[MONITOR_CURRENT_NEG]);
214 lcd->print(string_buf);
215 }

244 int16_t nearest10(int16_t value) {
245 int16_t a = (value / 10) * 10;
246 if ((value - a) >= 5) {
247 return a + 10;
248 }
249 return a;

262 char* generateVoltageString(int16_t value) {
263 // Generate the string from integer values

265 // Set seventh character to nul to terminate string
266 string_buf[6] = 0;
267 // Move rightmost 2 digits to the right

269 string_buf[4] = string_buf[3];
270 // Add decimal point
